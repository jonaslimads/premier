Aggregate roots:

* Vendor can create categories
* Vendor can add products linked to categories
[* Vendor can add menus. Menu can be child of a parent menu. Menu can be custom or of a category. If no menu is provided, categories will work as a menu.]

- Vendor
    id
    name
    isArchived: boolean?
    attributes: JSON?
    - categories: [
        id
        name
        slug
        isArchived: boolean?
        attributes: JSON?
        [- categories]
        - productIds: ID[]
    ]

- Product
    id
    vendorId
    name: String
    description: String
    slug: String
    currency: String
    price: Int
    cost: Int?
    - paymentOptions: [
        amount: Int,
        method: String (credit card, pix, boleto)
        discount: Int? (0),
        installments: Int? (1),
        expiresOn: DateTime?
    ]
    isArchived: boolean?
    attributes: JSON?
    images: []
    - variants: [
        id
        sku: String
        name: String?
        price: Int?
        cost: Int?
        - paymentOptions?: [
            amount: Int,
            method: String (credit card, pix, boleto)
            discount: Int? (0),
            installments: Int? (1),
            expiresOn: Date?
        ]
        isArchived: boolean?
        attributes: JSON?
        images: []
    ]

- Order
    id
    buyerId
    status
    currency
    totalGrossAmount: Int
    totalNetAmount: Int
    products: [
        id
        price
        - paymentOption
            amount: Int,
            method: String (credit card, pix, boleto)
            discount: Int? (0),
            installments: Int? (1)
    ]
    attributes: JSON?
    invoice: INVOICE
    events: [

    ]
